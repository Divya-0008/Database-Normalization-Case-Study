CREATE TABLE Employees_1NF (
 emp_id INT,

CREATE TABLE Employees (
 emp_id INT PRIMARY KEY,
 emp_name VARCHAR(100),
 department VARCHAR(100)
);

CREATE TABLE Projects (
 proj_id INT PRIMARY KEY,
 project_name VARCHAR(150)
);

CREATE TABLE EmployeeProjects (
 emp_id INT,
 proj_id INT,
 PRIMARY KEY (emp_id, proj_id),
 FOREIGN KEY (emp_id) REFERENCES Employees(emp_id),
 FOREIGN KEY (proj_id) REFERENCES Projects(proj_id)
);

CREATE TABLE Managers (
 mgr_id INT PRIMARY KEY,
 manager_name VARCHAR(100)
);

CREATE TABLE Projects_3NF (
 proj_id INT PRIMARY KEY,
 project_name VARCHAR(150),
 mgr_id INT,
 FOREIGN KEY (mgr_id) REFERENCES Managers(mgr_id)
);

SELECT e.emp_name, p.project_name, m.manager_name
FROM Employees e
JOIN EmployeeProjects ep ON e.emp_id = ep.emp_id
JOIN Projects_3NF p ON ep.proj_id = p.proj_id
JOIN Managers m ON p.mgr_id = m.mgr_id;

SHOW TABLES;
